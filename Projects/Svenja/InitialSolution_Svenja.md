\#trend analysis

    car_yearly <- car %>%
      group_by(year) %>%
      summarise(
        avg_mpg = mean(mpg, na.rm = TRUE),
        avg_hp = mean(horsepower, na.rm = TRUE),
        avg_acc = mean(acceleration, na.rm = TRUE)
      )

    p3 <- ggplot(car_yearly, aes(x = year)) +
      geom_line(aes(y = avg_mpg, color = "mpg"), size = 1) +
