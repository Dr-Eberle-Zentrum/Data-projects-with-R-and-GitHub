library(tidyverse)
library(readxl)
library(Hmisc)
Data <- read_xlsx("data.xlsx")
Data %>%
  rename(cat = countriesAndTerritories,
         population = popData2020,
         ccode=countryterritoryCode)%>%
  select(cases, deaths, cat, population, ccode)%>%
  filter(!(cat == "Iceland"| cat == "Liechtenstein"| cat == "Norway"))%>%
  dplyr::group_by(cat, ccode)%>%
  summarise(
            prop_deaths=(sum(deaths, na.rm= T)/population)*1000,
            prop_cases=(sum(cases, na.rm= T)/population)*1000) %>%
  distinct()%>%
  ungroup()-> mod_Data
`summarise()` has grouped output by 'cat', 'ccode'. You can override using the `.groups` argument.
mod_Data$Death_groups <- cut2(mod_Data$prop_deaths,g=3)
mod_Data$Death_groups <-factor(mod_Data$Death_groups, levels = rev(levels(mod_Data$Death_groups)))
