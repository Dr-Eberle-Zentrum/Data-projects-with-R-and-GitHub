data <- read_csv("cloze_task_results.csv", show_col_types = FALSE)
data %>%
  group_by(text)%>%
  summarise(
    key_word=text,
    response=response,
    sum_count = sum(count),
    count = count
  )%>%
filter(count==max(count, na.rm = T))%>%
mutate(prop_top_answ=count/sum_count)%>%
ungroup()%>%
  mutate(extracted=str_extract(text, "\\b(kein|keine)\\b\\s+\\w+"))->mod_data
